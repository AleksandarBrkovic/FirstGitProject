package scontro;

public class Arma {
	private int distanzaColpo;
	private int danni;
	private int livello;
	private String nome;
	private int difesa;
	private int dannoCritico;
	private int resistenza;
	
	public Arma() {
		danni=700;
		distanzaColpo=20;
		livello=1;
		nome="ExcaliZar";
		difesa= 25;
		dannoCritico= danni*9;
		resistenza=100;
		//monke like banana
	}
	
	public Arma(int d, int disC, int lvl, String n, int dif, int dC) {
		d=danni;
		disC=distanzaColpo;
		lvl=livello;
		n=nome;
		dif=difesa;
		dC=dannoCritico;
	}
	
	
	
	

}
